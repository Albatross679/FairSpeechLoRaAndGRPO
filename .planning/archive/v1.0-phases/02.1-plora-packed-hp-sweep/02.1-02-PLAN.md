def deactivate_adapter(self, name):
    """Mark adapter as pruned. Frees optimizer and data iterator but
    keeps adapter weights on model for potential post-hoc analysis."""
    if name not in self.active_adapters:
        return
    self.active_adapters.remove(name)
    # Free optimizer memory
    if name in self.optimizers:
        del self.optimizers[name]
    # Free data iterator
    if name in self.data_iters:
        del self.data_iters[name]
    gc.collect()
    if torch.cuda.is_available():
        torch.cuda.empty_cache()
    print(f"  Deactivated adapter {name}, {len(self.active_adapters)} active remaining")
```

def train(self, n_steps, eval_steps, rung_steps=None):
    effective_rungs = set(rung_steps) if rung_steps else self.rung_steps
    for step in range(1, n_steps + 1):
        if not self.active_adapters:
            print(f"  All adapters pruned at step {step}. Ending early.")
            break
        self.train_step(step)
        # Evaluate at regular intervals OR at rung steps
        if step % eval_steps == 0 or step == n_steps or step in effective_rungs:
            self.evaluate(step)
            if torch.cuda.is_available():
                vram_mb = torch.cuda.max_memory_allocated() / (1024**2)

                name: self.eval_losses[name][-1][1]
                for name in self.active_adapters
                if self.eval_losses[name]

                "active_count": len(self.active_adapters),

def _create_scheduler(self, optimizer, name, config, total_steps):
    sched_type = config.get("lr_scheduler", "constant")
    warmup_ratio = config.get("warmup_ratio", 0.0)
    warmup_steps = int(total_steps * warmup_ratio)

    if sched_type == "constant":
        return None

    def lr_lambda(current_step):
        if current_step < warmup_steps:
            return float(current_step) / float(max(1, warmup_steps))
        progress = float(current_step - warmup_steps) / float(max(1, total_steps - warmup_steps))
        if sched_type == "cosine":
            return max(0.0, 0.5 * (1.0 + math.cos(math.pi * progress)))
        else:  # linear
            return max(0.0, 1.0 - progress)

    return torch.optim.lr_scheduler.LambdaLR(optimizer, lr_lambda)
```

```python
def validate_gate(current_top3, prior_top3):
    """Gate: >= 2 shared entries matched by (rank, target_mlp)."""
    current_keys = {(c["params"]["rank"], c["params"]["target_mlp"]) for c in current_top3}
    prior_keys = {(p["params"]["rank"], p["params"]["target_mlp"]) for p in prior_top3}
    shared = current_keys & prior_keys
    passed = len(shared) >= 2
    return passed, list(shared), len(shared)
```
